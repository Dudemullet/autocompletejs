At it's most simple, AutoComplete works with an array of values.

var foods1 = ['Apple', 'Banana', 'Orange'];
var ac1 = new AutoComplete('#example1', foods1);

You can put values into groups.

var foods2 = [
  { group: 'Fruits', value: 'Apple' },
  { group: 'Fruits', value: 'Banana' },
  { group: 'Fruits', value: 'Orange' },
  { group: 'Meats', value: 'Beef' },
  { group: 'Meats', value: 'Chicken' },
  { group: 'Meats', value: 'Pork' },
  { group: 'Vegetables', value: 'Carrot' },
  { group: 'Vegetables', value: 'Lettuce' },
  { group: 'Vegetables', value: 'Onion' }
];
var ac2 = new AutoComplete('#example2', foods2);

You can select items from a key / value list.

var foods3 = [
  {
    value: 'Fruits',
    children: ['Apple', 'Banana', 'Orange']
  },
  {
    value: 'Meats',
    children: ['Beef', 'Chicken', 'Pork']
  },
  {
    value: 'Vegetables',
    children: ['Carrot', 'Lettuce', 'Onion']
  }
];
var ac3 = new AutoComplete('#example3', foods3);

Your lists can be arbitrarily nested.

var foods4 = [
  {
    value: 'Fruits',
    children: ['Apple', 'Banana', 'Orange']
  },
  {
    value: 'Meats',
    children: [
      {
        value: 'Beef',
        children: [
          'Hamburger',
          'Pot Roast',
          {
            value: 'Steak',
            children: [
              'Ribeye',
              'T-Bone'
            ]
          }
        ]
      },
      'Chicken',
      'Pork'
    ]
  },
  {
    value: 'Vegetables',
    children: ['Carrot', 'Lettuce', 'Onion']
  }
];
var ac4 = new AutoComplete('#example4', foods4);

You can allow or disallow free-form input.

var opts5 = {
  values: [
    {
      value: "State",
      children: ['Alabama', 'Alaska', 'Arizona']
    },
    {
      value: "Street",
      children: {
        allowFreeform: true
      }
    },
    {
      value: "Zip Code",
      children: {
        allowFreeform: true
      }
    }
  ]
};
var ac5 = new AutoComplete('#example5', opts5);
